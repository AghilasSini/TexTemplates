    \styleRub{#2}
    \setlength{\lenB}{#1}
    \setlength{\lenC}{\linewidth}
    \addtolength{\lenC}{-\lenA}
    \addtolength{\lenC}{-\lenB}
    \addtolength{\lenC}{-\parindent}
    \addtolength{\lenC}{-9pt}
